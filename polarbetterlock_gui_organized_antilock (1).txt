local function New(class, props, parent)
    local obj = Instance.new(class)
    for k, v in pairs(props or {}) do
        obj[k] = v
    end
    if parent then
        obj.Parent = parent
    end
    return obj
end

local function Round(parent, radius)
    New("UICorner", { CornerRadius = UDim.new(0, radius or 8) }, parent)
end

local function Stroke(parent, color, thickness, transparency)
    return New("UIStroke", {
        Color           = color or INNER_STROKE,
        Thickness       = thickness or 1.2,
        Transparency    = transparency or 0.35,
        ApplyStrokeMode = Enum.ApplyStrokeMode.Border,
    }, parent)
end

local function Tween(inst, info, props)

    t:Play()
    return t
end

local function GetKeyName(keyCode)
    local name = keyCode and keyCode.Name or "-"
    if name == "Unknown" then return "-" end
    if name == "ButtonA" then return "A" end
    if name == "ButtonB" then return "B" end
    if name == "ButtonX" then return "X" end
    if name == "ButtonY" then return "Y" end
    if name == "ButtonL1" then return "L1" end
    if name == "ButtonR1" then return "R1" end
    if name == "ButtonL2" then return "L2" end
    if name == "ButtonR2" then return "R2" end
    if name == "ButtonL3" then return "L3" end
    if name == "ButtonR3" then return "R3" end
    if name == "DPadUp" then return "DUP" end
    if name == "DPadDown" then return "DDN" end
    if name == "DPadLeft" then return "DLT" end
    if name == "DPadRight" then return "DRT" end
    if name == "LeftControl" then return "CTRL" end
    if name == "RightControl" then return "RCTRL" end
    if name == "LeftShift" then return "SHIFT" end
    if name == "RightShift" then return "RSHIFT" end
    if name == "Space" then return "SPACE" end
    return name:sub(1, 5):upper()
end

local function IsBindableInput(input)
    return input.UserInputType == Enum.UserInputType.Keyboard
        or input.UserInputType == Enum.UserInputType.Gamepad1
end

local function GetBindableKey(input)
    if not IsBindableInput(input) then return nil end
    if input.KeyCode == Enum.KeyCode.Unknown then return nil end
    return input.KeyCode
end

local function serializeUDim2(value)
    if typeof(value) ~= "UDim2" then return nil end
    return {
        XS = value.X.Scale,
        XO = value.X.Offset,
        YS = value.Y.Scale,
        YO = value.Y.Offset,
    }
end

local function deserializeUDim2(value, fallback)
    if type(value) ~= "table" then return fallback end
    return UDim2.new(
        tonumber(value.XS) or fallback.X.Scale,
        tonumber(value.XO) or fallback.X.Offset,
        tonumber(value.YS) or fallback.Y.Scale,
        tonumber(value.YO) or fallback.Y.Offset
    )
end

        AutoLeft = State.AutoLeft,
        AutoRight = State.AutoRight,
        AutoBat = State.AutoBat,
        BatAimbot = State.BatAimbot,
        AntiLock = State.AntiLock,
        BatCounter = State.BatCounter,
        FPSBooster = State.FPSBooster,
        HideButtons = State.HideButtons,
        MobileButtonsPosition = serializeUDim2(State.MobileButtonsPosition),
        Keys = {},
    }

    for name, keyCode in pairs(Keys) do
        cfg.Keys[name] = keyCode.Name
    end

    local ok, encoded = pcall(function()
        return HttpService:JSONEncode(cfg)
    end)
    if ok and encoded then
        pcall(function() _writefile(CONFIG_FILE, encoded) end)
    end
end

local function loadConfig()
    local exists = false
    pcall(function() exists = _isfile(CONFIG_FILE) end)
    if not exists then return end

    local raw
    pcall(function() raw = _readfile(CONFIG_FILE) end)
    if not raw then return end

    local cfg
    local ok = pcall(function()
        cfg = HttpService:JSONDecode(raw)
    end)
    if not ok or not cfg then return end

    for _, name in ipairs({
        "NormalSpeed", "CarrySpeed", "LaggerSpeed", "AimbotSpeed", "AimbotPrediction", "AimbotHeightOffset", "FOV",
        "StealRadius", "StealDuration", "CarryMode", "AutoCarrySpeed", "LaggerMode", "Unwalk", "DarkMode",
        "AutoSteal", "AntiRagdoll", "InfiniteJump", "MedusaCounter",
        "AutoLeft", "AutoRight", "AutoBat", "BatAimbot", "AntiLock", "BatCounter", "FPSBooster", "HideButtons",
    }) do
        if cfg[name] ~= nil then
            State[name] = cfg[name]
        end
    end

    if cfg.Keys then
        for name, keyName in pairs(cfg.Keys) do
            if Keys[name] and Enum.KeyCode[keyName] then
                Keys[name] = Enum.KeyCode[keyName]
            end
        end
    end

    State.MobileButtonsPosition = deserializeUDim2(cfg.MobileButtonsPosition, DEFAULT_MOBILE_BUTTONS_POSITION)
end

local function ShowNotif(icon, message, kind)
    kind = kind or "success"

    local accentColor = kind == "error" and Color3.fromRGB(255, 100, 100)
        or kind == "info" and T.LightBlue
        or T.Cyan

    local slotY = _notifSlotY + (#_notifQueue * (_notifHeight + _notifGap))

    local frame = New("Frame", {
        Size                   = UDim2.new(0, 280, 0, 56),
        Position               = UDim2.new(0, -380, 0, slotY),
        BackgroundColor3       = T.BG_Panel,
        BackgroundTransparency = 0.06,
        BorderSizePixel        = 0,
        ClipsDescendants       = true,
    }, NotifGui)

    Round(frame, 10)

    local brd = Stroke(frame, accentColor, 1.2, 0)

    New("UIGradient", {
        Rotation = 90,
        Color = ColorSequence.new({
            ColorSequenceKeypoint.new(0, accentColor),
            ColorSequenceKeypoint.new(1, T.White),
        }),
    }, brd)

    local glow = New("Frame", {
        Size             = UDim2.new(0, 80, 1, 0),
        BackgroundColor3 = accentColor,
        BorderSizePixel  = 0,
    }, frame)

    New("UIGradient", {
        Transparency = NumberSequence.new({
            NumberSequenceKeypoint.new(0, 0.65),
            NumberSequenceKeypoint.new(1, 1),
        }),
    }, glow)

    local iBg = New("Frame", {
        Position               = UDim2.new(0, 12, 0.5, -16),
        Size                   = UDim2.new(0, 32, 0, 32),
        BackgroundColor3       = accentColor,
        BackgroundTransparency = 0.7,
        BorderSizePixel        = 0,
    }, frame)

    Round(iBg, 6)

    New("TextLabel", {
        Size                   = UDim2.new(1, 0, 1, 0),
        BackgroundTransparency = 1,
        Text                   = icon,
        TextColor3             = accentColor,
        Font                   = Enum.Font.GothamBold,
        TextSize               = 16,
    }, iBg)

    New("TextLabel", {
        Position               = UDim2.new(0, 52, 0, 6),
        Size                   = UDim2.new(1, -60, 1, -12),
        BackgroundTransparency = 1,
        Text                   = message,
        TextColor3             = T.Text,
        Font                   = Enum.Font.GothamMedium,
        TextSize               = 13,
        TextXAlignment         = Enum.TextXAlignment.Left,
        TextYAlignment         = Enum.TextYAlignment.Center,
        TextWrapped            = true,
    }, frame)

    local prog = New("Frame", {
        Position         = UDim2.new(0, 0, 1, -2),
        Size             = UDim2.new(1, 0, 0, 2),
        BackgroundColor3 = accentColor,
        BorderSizePixel  = 0,
    }, frame)

    New("UIGradient", {
        Color = ColorSequence.new({
            ColorSequenceKeypoint.new(0, accentColor),
            ColorSequenceKeypoint.new(0.5, T.White),
            ColorSequenceKeypoint.new(1, accentColor),
        }),
    }, prog)

    table.insert(_notifQueue, frame)

    TweenService:Create(frame, TweenInfo.new(0.35, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
        Position = UDim2.new(0, 16, 0, slotY)
    }):Play()

    task.delay(0.4, function()
        TweenService:Create(prog, TweenInfo.new(3.1, Enum.EasingStyle.Linear), {
            Size = UDim2.new(0, 0, 0, 2)
        }):Play()
    end)

    task.delay(3.8, function()
        TweenService:Create(frame, TweenInfo.new(0.3, Enum.EasingStyle.Quart, Enum.EasingDirection.In), {
            Position = UDim2.new(0, -380, 0, slotY)
        }):Play()

        task.delay(0.35, function()
            frame:Destroy()
            local idx = table.find(_notifQueue, frame)
            if idx then
                table.remove(_notifQueue, idx)
            end
        end)
    end)
end

local function MakeSnowflake(size, parent)
    local flake = New("Frame", {
        AnchorPoint            = Vector2.new(0.5, 0.5),
        Size                   = UDim2.new(0, size, 0, size),
        BackgroundTransparency = 1,
        BorderSizePixel        = 0,
        ZIndex                 = 2,
    }, parent)

    local th = math.max(1, math.floor(size / 10))
    local arm = math.floor(size * 0.72)
    local br = math.max(3, math.floor(size * 0.22))

    for _, r in ipairs({0, 60, 120}) do
        New("Frame", {
            AnchorPoint            = Vector2.new(0.5, 0.5),
            Position               = UDim2.new(0.5, 0, 0.5, 0),
            Size                   = UDim2.new(0, arm, 0, th),
            BackgroundColor3       = T.Snowflake,
            BackgroundTransparency = 0.45,
            BorderSizePixel        = 0,
            Rotation               = r,
            ZIndex                 = 2,
        }, flake)
    end

    for _, r in ipairs({30, 90, 150, 210, 270, 330}) do
        New("Frame", {
            AnchorPoint            = Vector2.new(0.5, 0.5),
            Position               = UDim2.new(0.5, 0, 0.5, 0),
            Size                   = UDim2.new(0, br, 0, th),
            BackgroundColor3       = T.Snowflake,
            BackgroundTransparency = 0.55,
            BorderSizePixel        = 0,
            Rotation               = r,
            ZIndex                 = 2,
        }, flake)
    end

    return flake
end

    for _, def in ipairs(TAB_DEFS) do

            btn.BackgroundTransparency = isActive and 0.05 or 0.95

            if btn:FindFirstChild("UIGradient") then
                btn.UIGradient.Enabled = isActive
            end

            local ab = btn:FindFirstChild("AccentBar")
            if ab then
                ab.BackgroundColor3 = def.accent
                ab.BackgroundTransparency = isActive and 0 or 1

                    BackgroundTransparency = isActive and 0.4 or 0.82
                })
            end
        end

            content.Visible = isActive
        end
    end

    activeTab = name
end

local function stopCurrentAnimations()

        pcall(function()

    end

    end
end

local function clearRadiusCircle()
    for _, part in ipairs(radiusCircleParts) do
        if part and part.Parent then
            part:Destroy()
        end
    end
    radiusCircleParts = {}
end

local function createRadiusCircle(char)
    clearRadiusCircle()
    local root = char and char:FindFirstChild("HumanoidRootPart")
    if not root then return end

    local radius = math.max(tonumber(Steal.StealRadius) or tonumber(State.StealRadius) or 20, 1)
    local circumference = 2 * math.pi * radius
    for i = 1, RADIUS_PARTS do
        local part = Instance.new("Part")
        part.Name = "PolarStealRadiusRing"
        part.Anchored = true
        part.CanCollide = false
        part.CanTouch = false
        part.CanQuery = false
        part.CastShadow = false
        part.Material = Enum.Material.Neon
        part.Color = T.LightBlue
        part.Transparency = 0.08
        part.Size = Vector3.new(circumference / RADIUS_PARTS, 0.16, 0.34)
        part.Parent = workspace
        table.insert(radiusCircleParts, part)
    end
end

local function updateRadiusCircle()
    local char = LocalPlayer.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    if not root then
        clearRadiusCircle()
        return
    end

    if #radiusCircleParts ~= RADIUS_PARTS then
        createRadiusCircle(char)
    end

    local radius = math.max(tonumber(Steal.StealRadius) or tonumber(State.StealRadius) or 20, 1)
    local circumference = 2 * math.pi * radius
    for i, part in ipairs(radiusCircleParts) do
        if part and part.Parent then
            local angle = (i - 1) * (2 * math.pi / RADIUS_PARTS)
            local nextAngle = i * (2 * math.pi / RADIUS_PARTS)
            local midAngle = (angle + nextAngle) / 2
            local pos = root.Position + Vector3.new(math.cos(midAngle) * radius, -2.85, math.sin(midAngle) * radius)
            part.Color = T.LightBlue
            part.Transparency = State.AutoSteal and 0.06 or 0.14
            part.Size = Vector3.new(circumference / RADIUS_PARTS, 0.16, 0.34)
            part.CFrame = CFrame.new(pos) * CFrame.Angles(0, -midAngle, 0)
        end
    end
end

local function isMyPlotByName(n)
    local ct = tick()
    if Steal.plotCache[n] and (ct - (Steal.plotCacheTime[n] or 0)) < 2 then return Steal.plotCache[n] end
    local plots = workspace:FindFirstChild("Plots")
    if not plots then Steal.plotCache[n] = false; Steal.plotCacheTime[n] = ct; return false end
    local plot = plots:FindFirstChild(n)
    if not plot then Steal.plotCache[n] = false; Steal.plotCacheTime[n] = ct; return false end
    local sign = plot:FindFirstChild("PlotSign")
    if sign then
        local yb = sign:FindFirstChild("YourBase")
        if yb and yb:IsA("BillboardGui") then
            local r = yb.Enabled == true
            Steal.plotCache[n] = r; Steal.plotCacheTime[n] = ct; return r
        end
    end
    Steal.plotCache[n] = false; Steal.plotCacheTime[n] = ct; return false
end

local function findNearestPrompt()
    local char = LocalPlayer.Character; if not char then return nil end
    local root = char:FindFirstChild("HumanoidRootPart"); if not root then return nil end
    local ct = tick()
    if ct - Steal.promptCacheTime < 0.15 and #Steal.cachedPrompts > 0 then
        local np, nd = nil, math.huge
        for _, d in ipairs(Steal.cachedPrompts) do
            if d.spawn then
                local dist = (d.spawn.Position - root.Position).Magnitude
                if dist <= Steal.StealRadius and dist < nd then np = d.prompt; nd = dist end
            end
        end
        if np then return np end
    end
    Steal.cachedPrompts = {}; Steal.promptCacheTime = ct
    local plots = workspace:FindFirstChild("Plots"); if not plots then return nil end
    local np, nd = nil, math.huge
    for _, plot in ipairs(plots:GetChildren()) do
        if isMyPlotByName(plot.Name) then continue end
        local pods = plot:FindFirstChild("AnimalPodiums"); if not pods then continue end
        for _, pod in ipairs(pods:GetChildren()) do
            pcall(function()
                local base = pod:FindFirstChild("Base")
                local sp = base and (base:FindFirstChild("SpawnPoint") or base:FindFirstChild("Spawn"))
                if sp then
                    local att = sp:FindFirstChild("PromptAttachment")
                    if att then
                        for _, child in ipairs(att:GetChildren()) do
                            if child:IsA("ProximityPrompt") then
                                local dist = (sp.Position - root.Position).Magnitude
                                table.insert(Steal.cachedPrompts, {prompt = child, spawn = sp})
                                if dist <= Steal.StealRadius and dist < nd then np = child; nd = dist end
                                break
                            end
                        end
                    end
                end
            end)
        end
    end
    return np
end

local function resetProgressBar()
    setProgress(0)
end

local function setProgressBarValue(v)
    setProgress(math.clamp(v or 0, 0, 1))
end

local function disconnectAutoStealProgress()
    if combatConns.AutoStealProgress then
        combatConns.AutoStealProgress:Disconnect()
        combatConns.AutoStealProgress = nil
    end
end

local function resetAutoStealAttempt(data)
    disconnectAutoStealProgress()
    resetProgressBar()
    if data then data.ready = true end
    State.isStealing = false
    autoStealIsStealing = false
end

local function executeSteal(prompt)
    local ct = tick()
    if autoStealIsStealing then return end
    if isCarryingBrainrot and isCarryingBrainrot(LocalPlayer.Character) then
        State.AutoStealPausedForCarry = true
        State.isStealing = false
        if enableCarrySpeedFromPickup then enableCarrySpeedFromPickup() end
        resetProgressBar()
        return
    end
    if Steal.Data[prompt] and not pcall(function() return prompt.Parent end) then Steal.Data[prompt] = nil end
    if not Steal.Data[prompt] then
        Steal.Data[prompt] = {hold = {}, trigger = {}, ready = true}
        pcall(function()
            if getconnections then
                for _, c in ipairs(getconnections(prompt.PromptButtonHoldBegan)) do
                    if c.Function then table.insert(Steal.Data[prompt].hold, c.Function) end
                end
                for _, c in ipairs(getconnections(prompt.Triggered)) do
                    if c.Function then table.insert(Steal.Data[prompt].trigger, c.Function) end
                end
            else
                Steal.Data[prompt].useFallback = true
            end
        end)
    end
    local data = Steal.Data[prompt]; if not data.ready then return end
    data.ready = false; autoStealIsStealing = true; State.isStealing = true; State.stealStartTime = ct; State.lastStealTick = ct
    autoStealAttemptToken += 1
    local token = autoStealAttemptToken
    resetProgressBar()
    disconnectAutoStealProgress()
    combatConns.AutoStealProgress = RunService.Heartbeat:Connect(function()
        if not autoStealIsStealing or token ~= autoStealAttemptToken then
            disconnectAutoStealProgress()
            return
        end
        local prog = math.clamp((tick() - State.stealStartTime) / Steal.StealDuration, 0, 1)
        setProgressBarValue(prog)
    end)
    task.spawn(function()
        local ok = false
        pcall(function()
            if not data.useFallback then
                for _, fn in ipairs(data.hold) do task.spawn(fn) end
                task.wait(Steal.StealDuration)
                for _, fn in ipairs(data.trigger) do task.spawn(fn) end
                ok = true
            end
        end)
        if not ok and fireproximityprompt then
            pcall(function() fireproximityprompt(prompt); ok = true end)
        end
        if not ok then
            pcall(function() prompt:InputHoldBegin(); task.wait(Steal.StealDuration); prompt:InputHoldEnd() end)
        end
        if token ~= autoStealAttemptToken then return end
        disconnectAutoStealProgress()
        resetProgressBar()
        if isCarryingBrainrot and isCarryingBrainrot(LocalPlayer.Character) then
            State.AutoStealPausedForCarry = true
            if enableCarrySpeedFromPickup then enableCarrySpeedFromPickup() end
            resetAutoStealAttempt(data)
            return
        end
        State.WaitingForCarryPickup = false
        State.CarryPickupGraceUntil = 0
        resetAutoStealAttempt(data)
    end)
end

function startAutoSteal()
    if combatConns.AutoSteal then
        combatConns.AutoSteal:Disconnect()
        combatConns.AutoSteal = nil
    end
    disconnectAutoStealProgress()
    autoStealAttemptToken += 1
    autoStealIsStealing = false
    State.isStealing = false
    State.WaitingForCarryPickup = false
    State.CarryPickupGraceUntil = 0
    resetProgressBar()
    combatConns.AutoSteal = RunService.Heartbeat:Connect(function()
        if not Steal.AutoStealEnabled or autoStealIsStealing then return end
        if isCarryingBrainrot and isCarryingBrainrot(LocalPlayer.Character) then
            State.AutoStealPausedForCarry = true
            resetProgressBar()
            if enableCarrySpeedFromPickup then enableCarrySpeedFromPickup() end
            return
        end
        if State.AutoStealPausedForCarry then
            State.AutoStealPausedForCarry = false
            Steal.cachedPrompts = {}
            for _, data in pairs(Steal.Data) do
                if type(data) == "table" then data.ready = true end
            end
        end
        local p = findNearestPrompt(); if p then executeSteal(p) end
    end)
    autoStealHeartbeat = combatConns.AutoSteal
end

function stopAutoSteal()
    if combatConns.AutoSteal then combatConns.AutoSteal:Disconnect(); combatConns.AutoSteal = nil end
    disconnectAutoStealProgress()
    autoStealHeartbeat = nil
    autoStealIsStealing = false
    autoStealAttemptToken += 1
    State.isStealing = false
    State.AutoStealPausedForCarry = false
    State.WaitingForCarryPickup = false
    State.CarryPickupGraceUntil = 0
    State.lastStealTick = 0
    Steal.plotCache = {}; Steal.plotCacheTime = {}; Steal.cachedPrompts = {}; Steal.Data = {}
    resetProgressBar()
end

                or state == Enum.HumanoidStateType.FallingDown then
                pcall(function() humanoid:ChangeState(Enum.HumanoidStateType.GettingUp) end)

                root.RotVelocity = Vector3.new(0, 0, 0)
            end

            for _, obj in ipairs(char:GetDescendants()) do
                if obj:IsA("Motor6D") and obj.Enabled == false then
                    pcall(function() obj.Enabled = true end)
                end
            end
        end)
    end

local function attachAntiDieHumanoid(char, humanoid, token)
    if token ~= antiDieToken or not char or not char.Parent or not humanoid or not humanoid.Parent then
        return
    end

    clearAntiDieConns()
    antiDieHumanoid = humanoid

    pcall(function()
        humanoid.BreakJointsOnDeath = false
        humanoid:SetStateEnabled(Enum.HumanoidStateType.Dead, false)
    end)

    setAntiDieCameraSubject(humanoid)

    antiDieConns.Health = humanoid:GetPropertyChangedSignal("Health"):Connect(function()
        if token ~= antiDieToken or antiDieHumanoid ~= humanoid then
            return
        end
        if humanoid.Health <= 0 then
            humanoid.Health = humanoid.MaxHealth
        end
    end)

    antiDieConns.Died = humanoid.Died:Connect(function()
        if token ~= antiDieToken or antiDieHumanoid ~= humanoid or antiDieReplacing then
            return
        end

        antiDieReplacing = true
        task.defer(function()
            if token ~= antiDieToken or not char.Parent or antiDieHumanoid ~= humanoid then
                antiDieReplacing = false
                return
            end

            local replacement = char:FindFirstChild("ReplacedHumanoid")
            if replacement and not replacement:IsA("Humanoid") then
                replacement = nil
            end
            if not replacement then
                replacement = Instance.new("Humanoid")
                replacement.Name = "ReplacedHumanoid"
                replacement.Parent = char

        end)
    end)

        if token == antiDieToken and antiDieHumanoid == humanoid and not parent and not antiDieReplacing then
            clearAntiDieConns()
        end
    end)
end

    State.FPSBooster = on

        miscConns.FPSBooster:Disconnect()
        miscConns.FPSBooster = nil
    end
    if not on then return end

    local function optimize(obj)
        pcall(function()
            if obj:IsA("BasePart") then
                obj.CastShadow = false
                obj.Material = Enum.Material.Plastic
                obj.Reflectance = 0

                obj.CastShadow = false

        if State.FPSBooster then task.defer(optimize, obj) end
    end)

local function runAutoPath(side)
    local phaseKey = side == "Left" and "AutoLeftPhase" or "AutoRightPhase"
    local enabledKey = side == "Left" and "AutoLeft" or "AutoRight"
    local stopFn = side == "Left" and stopAutoLeft or stopAutoRight
    local faceYaw = side == "Left" and 0 or math.rad(180)
    local first = side == "Left" and POS.L1 or POS.R1
    local second = side == "Left" and POS.L2 or POS.R2
    local connKey = enabledKey

    if combatConns[connKey] then combatConns[connKey]:Disconnect() end
    State[phaseKey] = 1
    combatConns[connKey] = RunService.Heartbeat:Connect(function()
        if not State[enabledKey] or State.CountdownActive then return end
        local char = LocalPlayer.Character
        local root = char and char:FindFirstChild("HumanoidRootPart")
        local hum = char and char:FindFirstChildOfClass("Humanoid")
        if not root or not hum then return end

        local target = State[phaseKey] == 1 and first or second
        local delta = Vector3.new(target.X, root.Position.Y, target.Z) - root.Position
        if delta.Magnitude < 1 then
            if State[phaseKey] == 1 then
                State[phaseKey] = 2
            else
                State[enabledKey] = false
                if toggleRefs[enabledKey] then toggleRefs[enabledKey](false) end
                if mobileSetters[enabledKey] then mobileSetters[enabledKey](false) end
                saveConfig()
                stopFn()
                root.CFrame = CFrame.new(root.Position) * CFrame.Angles(0, faceYaw, 0)
            end
            return
        end

        local move = Vector3.new(delta.X, 0, delta.Z).Unit
        hum:Move(move, false)
        root.Velocity = Vector3.new(move.X * State.NormalSpeed, root.Velocity.Y, move.Z * State.NormalSpeed)
    end)
end

-- NEW BAT AIMBOT (lock-on movement, prediction, height offset)
-- ============================================================
local batAimbotConn = nil
local batAimbotCurrentTarget = nil

local function isValidCharacter(char)
    if not char then return false end
    local hum = char:FindFirstChildOfClass("Humanoid")
    local hrp = char:FindFirstChild("HumanoidRootPart")
    return hum and hrp and hum.Health > 0
end

local function getClosestTarget(pos)
    local closest, bestDist = nil, math.huge
    for _, plr in pairs(Players:GetPlayers()) do
        if plr ~= LocalPlayer and isValidCharacter(plr.Character) then
            local hrp = plr.Character.HumanoidRootPart
            local mag = (hrp.Position - pos).Magnitude
            if mag < bestDist then
                bestDist = mag
                closest = plr.Character
            end
        end
    end
    return closest
end

local function onBatAimbotRender()
    if not State.BatAimbot then return end

    local char = LocalPlayer.Character
    if not isValidCharacter(char) then return end

    local hrp = char.HumanoidRootPart
    local hum = char:FindFirstChildOfClass("Humanoid")
    hum.AutoRotate = false

    if batAimbotCurrentTarget and not isValidCharacter(batAimbotCurrentTarget) then
        batAimbotCurrentTarget = nil
    end

    if not batAimbotCurrentTarget then
        batAimbotCurrentTarget = getClosestTarget(hrp.Position)
    end

    if not batAimbotCurrentTarget then
        hrp.Velocity = Vector3.zero
        return
    end

    local targetHRP = batAimbotCurrentTarget.HumanoidRootPart
    local predicted = targetHRP.Position + (targetHRP.Velocity * State.AimbotPrediction)
    local standPos = predicted + Vector3.new(0, State.AimbotHeightOffset, 0)
    local moveDir = standPos - hrp.Position

    if moveDir.Magnitude > 1 then
        hrp.Velocity = moveDir.Unit * State.AimbotSpeed
    else
        hrp.Velocity = Vector3.zero
    end

    local flatLook = Vector3.new(predicted.X, hrp.Position.Y, predicted.Z)
    hrp.CFrame = CFrame.lookAt(hrp.Position, flatLook)
end

local function startBatAimbot()
    if batAimbotConn then
        batAimbotConn:Disconnect()
        batAimbotConn = nil
    end
    batAimbotCurrentTarget = nil
    batAimbotConn = RunService.RenderStepped:Connect(onBatAimbotRender)
end

local function stopBatAimbot()
    if batAimbotConn then
        batAimbotConn:Disconnect()
        batAimbotConn = nil
    end
    batAimbotCurrentTarget = nil

    local char = LocalPlayer.Character
    if char then
        local hum = char:FindFirstChildOfClass("Humanoid")
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if hum then hum.AutoRotate = true end
        if hrp then hrp.Velocity = Vector3.zero end
    end
end

local batSwingCooldown = false
local function findBatTool()
    local char = LocalPlayer.Character
    if char then
        for _, child in ipairs(char:GetChildren()) do
            if child:IsA("Tool") and child.Name:lower():find("bat") then return child end
        end
    end
    local backpack = LocalPlayer:FindFirstChildOfClass("Backpack")
    if backpack then
        for _, child in ipairs(backpack:GetChildren()) do
            if child:IsA("Tool") and child.Name:lower():find("bat") then return child end
        end
    end
    return nil
end

local function swingBat()
    if batSwingCooldown then return end
    local bat = findBatTool()
    if not bat then return end
    batSwingCooldown = true
    pcall(function() bat:Activate() end)
    task.delay(0.16, function()
        batSwingCooldown = false
    end)
end

local autoBatStep = 0
local autoBatWasSwinging = false

local function pauseAutoBatSwing()
    autoBatStep = 0
    autoBatWasSwinging = false

    local char = LocalPlayer.Character
    local bat = char and char:FindFirstChild("Bat")
    if bat and bat:IsA("Tool") then
        pcall(function() bat:Deactivate() end)
    end
end

local function stopAutoBat()
    if combatConns.AutoBat then
        combatConns.AutoBat:Disconnect()
        combatConns.AutoBat = nil
    end
    pauseAutoBatSwing()
end

local function startAutoBat()
    stopAutoBat()

    combatConns.AutoBat = RunService.Heartbeat:Connect(function(dt)
        if not State.AutoBat then
            stopAutoBat()
            return
        end

        if not State.BatAimbot then
            if autoBatWasSwinging then
                pauseAutoBatSwing()
            end
            return
        end

        autoBatStep += dt
        if autoBatStep < 0.12 then
            return
        end
        autoBatStep = 0
        autoBatWasSwinging = true

        local char = LocalPlayer.Character
        if not char then return end

        local humanoid = char:FindFirstChildOfClass("Humanoid")
        if not humanoid or humanoid.Health <= 0 then return end

        local bat = char:FindFirstChild("Bat")
        if not bat then
            local backpack = LocalPlayer:FindFirstChildOfClass("Backpack")
            bat = backpack and backpack:FindFirstChild("Bat")
            if bat and bat:IsA("Tool") then
                pcall(function() humanoid:EquipTool(bat) end)
            end
        end

        bat = char:FindFirstChild("Bat")
        if bat and bat:IsA("Tool") then
            pcall(function() bat:Activate() end)
        end
    end)
end

local function findMedusaTool()
    local char = LocalPlayer.Character
    local backpack = LocalPlayer:FindFirstChildOfClass("Backpack")
    for _, container in ipairs({ char, backpack }) do
        if container then
            for _, item in ipairs(container:GetChildren()) do
                local name = item.Name:lower()
                if item:IsA("Tool") and (name:find("medusa") or name:find("head") or name:find("stone")) then
                    if container == backpack and char then
                        local hum = char:FindFirstChildOfClass("Humanoid")
                        if hum then pcall(function() hum:EquipTool(item) end) end
                    end
                    return item
                end
            end
        end
    end
    return nil
end

local function setCombatMode(key, on)
    State[key] = on
    if mobileSetters[key] then mobileSetters[key](on) end
    saveConfig()

    if key == "AutoLeft" and on and State.BatAimbot then
        State.BatAimbot = false
        if toggleRefs.BatAimbot then toggleRefs.BatAimbot(false) end
        if mobileSetters.BatAimbot then mobileSetters.BatAimbot(false) end
        stopBatAimbot()
        if State.AutoBat then
            pauseAutoBatSwing()
        end
    elseif key == "AutoRight" and on and State.BatAimbot then
        State.BatAimbot = false
        if toggleRefs.BatAimbot then toggleRefs.BatAimbot(false) end
        if mobileSetters.BatAimbot then mobileSetters.BatAimbot(false) end
        stopBatAimbot()
        if State.AutoBat then
            pauseAutoBatSwing()
        end
    elseif key == "BatAimbot" and on then
        if State.AutoLeft then
            State.AutoLeft = false
            if toggleRefs.AutoLeft then toggleRefs.AutoLeft(false) end
            if mobileSetters.AutoLeft then mobileSetters.AutoLeft(false) end
            stopAutoLeft()
        end
        if State.AutoRight then
            State.AutoRight = false
            if toggleRefs.AutoRight then toggleRefs.AutoRight(false) end
            if mobileSetters.AutoRight then mobileSetters.AutoRight(false) end
            stopAutoRight()
        end
    end

    if key == "AutoLeft" then
        if on then runAutoPath("Left") else stopAutoLeft() end
    elseif key == "AutoRight" then
        if on then runAutoPath("Right") else stopAutoRight() end
    elseif key == "BatAimbot" then
        if on then
            startBatAimbot()
        else
            stopBatAimbot()
            if State.AutoBat then
                pauseAutoBatSwing()
            end
        end
    end
end

local function runTPDown()
    local char = LocalPlayer.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    if not root then return end

    local excludeList = {}
    for _, part in ipairs(char:GetDescendants()) do
        if part:IsA("BasePart") then table.insert(excludeList, part) end
    end
    table.insert(excludeList, root)

    local params = RaycastParams.new()
    params.FilterDescendantsInstances = excludeList
    params.FilterType = Enum.RaycastFilterType.Exclude
    local hit = workspace:Raycast(root.Position + Vector3.new(0, -0.1, 0), Vector3.new(0, -2000, 0), params)
    local yaw = select(2, root.CFrame:ToEulerAnglesYXZ())
    if hit and hit.Position.Y > -200 then
        root.CFrame = CFrame.new(root.Position.X, hit.Position.Y + 3.1, root.Position.Z) * CFrame.Angles(0, yaw, 0)
        root.Velocity = Vector3.zero
    end
end

local function MakeDraggable(frame, checkLock, onDragEnded, dragHandle)
    local dragging, dragInput, dragStart, startPos
    local handle = dragHandle or frame

    handle.InputBegan:Connect(function(input)
        if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch)
            and UserInputService:GetFocusedTextBox() == nil then

            if checkLock and State.UiLocked then
                return
            end

            dragging = true
            dragInput = input.UserInputType == Enum.UserInputType.Touch and input or dragInput
            dragStart = input.Position
            startPos = frame.Position

            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                    if onDragEnded then
                        onDragEnded(frame.Position)
                    end
                end
            end)
        end
    end)

    handle.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and input == dragInput then
            if checkLock and State.UiLocked then
                dragging = false
                return
            end

            local delta = input.Position - dragStart

            frame.Position = UDim2.new(
                startPos.X.Scale,
                startPos.X.Offset + delta.X,
                startPos.Y.Scale,
                startPos.Y.Offset + delta.Y
            )
        end
    end)
end

local function isCarryItemName(name)
    local n = tostring(name or ""):lower()
    return n:find("brainrot") or n:find("animal") or n:find("carry")
        or n:find("grab") or n:find("steal") or n:find("hold")
end

local function isIgnoredCarryTool(name)
    local n = tostring(name or ""):lower()
    return n:find("bat") or n:find("slap") or n:find("medusa")
        or n:find("head") or n:find("stone")
end

isCarryingBrainrot = function(char)
    if not char then return false end

    for _, name in ipairs({ "Carrying", "IsCarrying", "Grabbed", "Holding", "StealHold", "HasGrab" }) do
        local value = char:FindFirstChild(name, true)
        if value then
            if value:IsA("BoolValue") and value.Value then return true end
            if value:IsA("ObjectValue") and value.Value then return true end
            if value:IsA("StringValue") and value.Value ~= "" then return true end
        end
    end

    for _, child in ipairs(char:GetChildren()) do
        if child:IsA("Model") and child:FindFirstChildWhichIsA("BasePart", true) then
            if child:FindFirstChildOfClass("Humanoid") and child:FindFirstChild("HumanoidRootPart") then return true end
            if isCarryItemName(child.Name) then return true end
        elseif child:IsA("Tool") and not isIgnoredCarryTool(child.Name) then
            return true
        end
    end

    return false
end

enableCarrySpeedFromPickup = function()
    if not State.AutoCarrySpeed then
        State.AutoCarryFromPickup = false
        State.WaitingForCarryPickup = false
        State.CarryPickupGraceUntil = 0

    end

    State.WaitingForCarryPickup = false
    State.AutoCarryFromPickup = true
    State.CarryPickupGraceUntil = tick() + 0.75
    setLaggerMode(false)
    setCarryMode(true)

disableCarrySpeedFromPickup = function()
    if not State.AutoCarryFromPickup and not State.WaitingForCarryPickup then return end
    State.AutoCarryFromPickup = false
    State.WaitingForCarryPickup = false
    State.CarryPickupGraceUntil = 0
    if State.AutoStealPausedForCarry then
        State.AutoStealPausedForCarry = false
        autoStealIsStealing = false
        State.isStealing = false
        State.lastStealTick = 0
        Steal.cachedPrompts = {}
        for _, data in pairs(Steal.Data) do
            if type(data) == "table" then data.ready = true end
        end
    end
    setLaggerMode(false)
    setCarryMode(false)
end

        setLaggerMode(not State.LaggerMode)

            stopUnwalk()

                toggleRefs.Unwalk(false)

        else
            startUnwalk()
            if toggleRefs.Unwalk then
                toggleRefs.Unwalk(true)

        end
    elseif bindKey == Keys.AutoLeft then
        local on = not State.AutoLeft
        if toggleRefs.AutoLeft then toggleRefs.AutoLeft(on) end

    elseif bindKey == Keys.AutoRight then

        local on = not State.BatAimbot
        if toggleRefs.BatAimbot then toggleRefs.BatAimbot(on) end
        setCombatMode("BatAimbot", on)
    elseif bindKey == Keys.AntiLock then
        setAntiLock(not State.AntiLock)
    elseif bindKey == Keys.DropBrainrot then
        State.DropBrainrot = true
        if toggleRefs.DropBrainrot then toggleRefs.DropBrainrot(true) end
        runDropBrainrot()
    elseif bindKey == Keys.TPDown then

        runTPDown()
    end
end)

-- RUNTIME LOOPS
local colorPhase = 0

local statusFrameCount, statusLastTick = 0, tick()
local statusPingStep = 0

local function startStatusStats()
    if miscConns.StatusFPS then
        miscConns.StatusFPS:Disconnect()
        miscConns.StatusFPS = nil
    end
    if miscConns.StatusPing then
        miscConns.StatusPing:Disconnect()
        miscConns.StatusPing = nil
    end

    statusFrameCount = 0
    statusLastTick = tick()
    statusPingStep = 0

    miscConns.StatusFPS = RunService.RenderStepped:Connect(function()
        statusFrameCount += 1
        local now = tick()
        if now - statusLastTick >= 0.5 then
            State.FPS = math.floor(statusFrameCount / (now - statusLastTick) + 0.5)
            statusFrameCount = 0
            statusLastTick = now
            updateStatusBar()
        end
    end)

    miscConns.StatusPing = RunService.Heartbeat:Connect(function(dt)
        statusPingStep += dt
        if statusPingStep >= 1 then
            statusPingStep = 0
            local ok, pingValue = pcall(function()
                return Stats.Network.ServerStatsItem["Data Ping"]:GetValue()
            end)
            if ok and pingValue then
                State.Ping = math.floor(pingValue + 0.5)
                updateStatusBar()
            end
        end
    end)
end

startStatusStats()

RunService.Stepped:Connect(function(_, dt)
    colorPhase = (colorPhase + dt * 40) % 360

    local breath = 0.22 + math.sin(colorPhase * 0.035) * 0.18
    local blend = (math.sin(colorPhase * 0.02) + 1) * 0.5

    local r = OUTER_STROKE.R * blend + INNER_STROKE.R * (1 - blend)
    local g = OUTER_STROKE.G * blend + INNER_STROKE.G * (1 - blend)
    local b = OUTER_STROKE.B * blend + INNER_STROKE.B * (1 - blend)

    bkBorder.Color = Color3.new(r, g, b)
    bkBorder.Transparency = breath
end)

RunService.RenderStepped:Connect(function()
    updateRadiusCircle()

    local char = LocalPlayer.Character
    local humanoid = char and char:FindFirstChildOfClass("Humanoid")
    local root = char and char:FindFirstChild("HumanoidRootPart")

    if humanoid and root then
        local carrying = isCarryingBrainrot(char) or LocalPlayer:GetAttribute("Stealing") == true
        local graceDone = tick() > (State.CarryPickupGraceUntil or 0)

        if State.AutoCarrySpeed and State.WaitingForCarryPickup then
            if carrying then
                enableCarrySpeedFromPickup()
            elseif graceDone then
                State.WaitingForCarryPickup = false
                if not State.AutoStealPausedForCarry then
                    autoStealIsStealing = false
                    State.isStealing = false
                end
            end
        end

        if State.AutoCarrySpeed and carrying and (not State.AutoCarryFromPickup or not State.CarryMode) then
            enableCarrySpeedFromPickup()
        elseif State.AutoCarryFromPickup and graceDone and not carrying then
            disableCarrySpeedFromPickup()
        end

        local moveDir = humanoid.MoveDirection
        local speed = State.LaggerMode and State.LaggerSpeed
            or State.CarryMode and State.CarrySpeed
            or State.NormalSpeed

        if moveDir.Magnitude > 0 then
            State.LastMoveDir = moveDir
            local currentVelocity = root.Velocity
            root.Velocity = Vector3.new(
                moveDir.X * speed,
                currentVelocity.Y,
                moveDir.Z * speed
            )
        end

        pcall(function()
            local head = char:FindFirstChild("Head")
            local bb = head and head:FindFirstChild("PolarDiscordTag")
            local speedLabel = bb and bb:FindFirstChild("SpeedBillLbl")
            if speedLabel then
                speedLabel.Text = string.format("%.1f", Vector3.new(root.Velocity.X, 0, root.Velocity.Z).Magnitude)
            end
        end)
    else
        disableCarrySpeedFromPickup()
    end

    pcall(function()
        if workspace.CurrentCamera and workspace.CurrentCamera.FieldOfView ~= State.FOV then
            workspace.CurrentCamera.FieldOfView = State.FOV
        end
    end)

    local t = tick()

    starLayer.Position = UDim2.new(
        0.5 + math.sin(t * 0.03) * 0.04,
        0,
        0.5 + math.cos(t * 0.025) * 0.04,
        0
    )

    for _, sd in ipairs(snowflakeData) do
        local flake = sd.flake

        SetSnowflakeTransparency(flake, 0.15 + math.abs(math.sin(t * sd.spd + sd.phase)) * 0.65)

        flake.Position = UDim2.new(
            sd.xBase + math.sin(t * sd.spd * 0.12 + sd.phase) * 0.03,
            0,
            sd.base + math.sin(t * sd.spd * 0.18 + sd.phase) * 0.07,
            0
        )

        flake.Rotation = flake.Rotation + sd.rotate * 0.01
    end
end)

local DISCORD_TAG = "/polarhub"

local function addDiscordTag(char)
    if not char then return end

    local head = char:WaitForChild("Head", 5)
    if not head then return end

    local old = head:FindFirstChild("PolarDiscordTag")
    if old then
        old:Destroy()
    end

    local bb = Instance.new("BillboardGui")
    bb.Name = "PolarDiscordTag"
    bb.Size = UDim2.new(0, 180, 0, 76)
    bb.StudsOffset = Vector3.new(0, 3, 0)
    bb.AlwaysOnTop = true
    bb.LightInfluence = 0
    bb.MaxDistance = 300
    bb.Parent = head

    local speedBillLbl = Instance.new("TextLabel")
    speedBillLbl.Name = "SpeedBillLbl"
    speedBillLbl.Size = UDim2.new(1, 0, 0, 24)
    speedBillLbl.Position = UDim2.new(0, 0, 0, 26)
    speedBillLbl.BackgroundTransparency = 1
    speedBillLbl.Text = "0.0"
    speedBillLbl.TextColor3 = Color3.fromRGB(190, 225, 255)
    speedBillLbl.Font = Enum.Font.GothamBlack
    speedBillLbl.TextScaled = true
    speedBillLbl.TextStrokeTransparency = 0.1
    speedBillLbl.TextStrokeColor3 = Color3.new(0, 0, 0)
    speedBillLbl.Parent = bb

    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(1, 0, 0, 24)
    label.Position = UDim2.new(0, 0, 0, 52)
    label.BackgroundTransparency = 1
    label.Text = DISCORD_TAG
    label.TextColor3 = Color3.fromRGB(255, 255, 255)
    label.TextStrokeTransparency = 0.1
    label.TextStrokeColor3 = Color3.new(0, 0, 0)
    label.Font = Enum.Font.GothamBold
    label.TextScaled = true
    label.Parent = bb
end
