local configFolder = "TricTracScanner"
local configPath = configFolder .. "/config.json"

local function saveConfig()
    pcall(function()
        if not isfolder(configFolder) then
            makefolder(configFolder)
        end
        local data = HttpService:JSONEncode({
            hopEnabled = hopEnabled,
            minValue = minValue
        })
        writefile(configPath, data)
    end)
end

local function loadConfig()
    pcall(function()
        if isfile(configPath) then
            local data = HttpService:JSONDecode(readfile(configPath))
            if type(data) == "table" then
                if data.hopEnabled ~= nil then hopEnabled = data.hopEnabled end
                if data.minValue ~= nil then minValue = data.minValue end
            end
        end
    end)
end

local ws = nil
local function connectWebSocket()
    if ws then return end
    local success, sock = pcall(function()
        if WebSocket then return WebSocket.connect(WS_URL) end
        if syn and syn.websocket then return syn.websocket.connect(WS_URL) end
        if fluxus and fluxus.websocket then return fluxus.websocket.connect(WS_URL) end
        if websocket and websocket.connect then return websocket.connect(WS_URL) end
        return nil
    end)
    if not success or not sock then return end
    ws = sock
    ws.OnClose:Connect(function()
        ws = nil
        task.wait(1)
        connectWebSocket()
    end)
end

local function sendWebhook(brainrotName, generation, playerCount, jobId)
    pcall(function()
        local joinLink = "https://www.roblox.com/start?placeId=" .. tostring(game.PlaceId) .. "&gameInstanceId=" .. tostring(jobId)
        
        local payload = HttpService:JSONEncode({
            username = "Tric Trac AJ",
            avatar_url = "",
            embeds = {{
                title = "TRIC TRAC SCANNER",
                color = 5763719,
                fields = {
                    { name = "🧠 Brainrot", value = tostring(brainrotName), inline = true },
                    { name = "💰 Generation", value = tostring(generation), inline = true },
                    { name = "👥 Players", value = tostring(playerCount), inline = true },
                    { name = "🆔 Job ID", value = "```" .. tostring(jobId) .. "```", inline = false },
                    { name = "🔗 Join Server", value = joinLink, inline = false },
                },
                footer = { text = "Tric Trac Scanner • Auto Hop" },
                timestamp = os.date("!%Y-%m-%dT%H:%M:%SZ"),
            }}
        })
        if syn and syn.request then
            syn.request({ Url = WEBHOOK_URL, Method = "POST", Headers = { ["Content-Type"] = "application/json" }, Body = payload })
        elseif http and http.request then
            http.request({ Url = WEBHOOK_URL, Method = "POST", Headers = { ["Content-Type"] = "application/json" }, Body = payload })
        elseif request then
            request({ Url = WEBHOOK_URL, Method = "POST", Headers = { ["Content-Type"] = "application/json" }, Body = payload })
        end
    end)
end

local function sendWebSocket(brainrotName, generation, playerCount, jobId)
    if not ws then return end
    pcall(function()
        local data = HttpService:JSONEncode({
            brainrot = brainrotName,
            generation = generation,
            players = playerCount,
            jobId = jobId or game.JobId
        })
        ws:Send(data)
    end)
end

local function getValue(text)
    if not text then return 0 end
    local num = tonumber(text:match("[%d%.]+")) or 0
    if text:find("[Mm]") then return num * 1000000
    elseif text:find("[Kk]") then return num * 1000
    else return num end
end

local function serverHop()
    local serverList = {}
    pcall(function()
        local url = "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Desc&limit=100"
        local res = game:HttpGet(url)
        local data = HttpService:JSONDecode(res)
        if data and data.data then
            for _, server in pairs(data.data) do
                if type(server) == "table" and server.id and server.id ~= game.JobId and server.playing and server.playing < server.maxPlayers then
                    if not table.find(visitedServers, server.id) then
                        table.insert(serverList, server.id)
                    end
                end
            end
        end
    end)
    
    if #serverList == 0 then
        visitedServers = {game.JobId}
        shared.TricTracVisited = visitedServers
        pcall(function()
            local url = "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"
            local res = game:HttpGet(url)
            local data = HttpService:JSONDecode(res)
            if data and data.data then
                for _, server in pairs(data.data) do
                    if type(server) == "table" and server.id and server.id ~= game.JobId and server.playing and server.playing < server.maxPlayers then
                        table.insert(serverList, server.id)
                    end
                end
            end
        end)
    end
    
    if #serverList > 0 then
        local chosenServer = serverList[math.random(1, #serverList)]
        table.insert(visitedServers, chosenServer)
        shared.TricTracVisited = visitedServers
        
        local queue = queue_on_teleport or (syn and syn.queue_on_teleport) or (fluxus and fluxus.queue_on_teleport)
        if queue then
            pcall(function()
                queue('shared.TricTracVisited = game:GetService("HttpService"):JSONDecode([====[' .. HttpService:JSONEncode(visitedServers) .. '====])')
            end)
        end
        
        TeleportService:TeleportToPlaceInstance(game.PlaceId, chosenServer, Player)
    else
        TeleportService:Teleport(game.PlaceId, Player)
    end
end

local function scan()
    local highest = 0
    local bestName = nil
    local bestGen = nil
    local bestJobId = game.JobId

    local debris = Workspace:FindFirstChild("Debris")
    if debris then
        for _, obj in ipairs(debris:GetChildren()) do
            local genLabel = obj:FindFirstChild("Generation", true) or obj:FindFirstChildWhichIsA("TextLabel")
            if genLabel and genLabel.Text and genLabel.Text:find("%$") then
                local value = getValue(genLabel.Text)
                if value > highest then
                    highest = value
                    bestGen = genLabel.Text
                    local display = obj:FindFirstChild("DisplayName", true) or obj:FindFirstChild("Displayname", true)
                    bestName = display and display.Text or "Unknown"
                    bestJobId = game.JobId
                end
            end
        end
    end

    local plots = Workspace:FindFirstChild("Plots")
    if plots then
        for _, plot in ipairs(plots:GetChildren()) do
            local podiums = plot:FindFirstChild("AnimalPodiums")
            if podiums then
                for _, pod in ipairs(podiums:GetChildren()) do
                    local gen = pod:FindFirstChild("Generation", true)
                    if gen and gen:IsA("TextLabel") and gen.Text:find("%$") then
                        local value = getValue(gen.Text)
                        if value > highest then
                            highest = value
                            bestGen = gen.Text
                            local display = pod:FindFirstChild("DisplayName", true) or pod:FindFirstChild("Displayname", true)
                            bestName = display and display.Text or "Podium Animal"
                            bestJobId = game.JobId
                        end
                    end
                end
            end
        end
    end

    if highest >= minValue then
        local logKey = bestName .. "_" .. bestGen .. "_" .. bestJobId
        
        if not loggedItems[logKey] then
            loggedItems[logKey] = true
            
            local players = #Players:GetPlayers() .. "/" .. Players.MaxPlayers
            sendWebhook(bestName, bestGen, players, bestJobId)
            sendWebSocket(bestName, bestGen, players, bestJobId)
            print(string.format("[TricTrac] Found %s (%s) - value %s", bestName, bestGen, highest))
        end
    end
end

    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true; dragStart = input.Position; startPos = Frame.Position
        input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false end end)
    end
end)
UserInputService.InputChanged:Connect(function(input)
    if dragging and dragStart and startPos and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - dragStart
        Frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset+delta.X, startPos.Y.Scale, startPos.Y.Offset+delta.Y)
    end
end)
