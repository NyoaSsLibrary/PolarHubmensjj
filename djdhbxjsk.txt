local function _llRequestFn()
	return (syn and syn.request)
		or (http and http.request)
		or http_request
		or request
end

local function _llUrlEncode(s)
	return (s:gsub("[^%w%-_%.~]", function(c)
		return string.format("%%%02X", string.byte(c))
	end))
end

local function _llWikiCandidates(displayName)
	local clean = displayName:gsub("%[.-%]%s*", ""):gsub("^%s+", ""):gsub("%s+$", "")
	clean = clean:match("^(.-)%s*%(") or clean
	local seen, list = {}, {}
	local function add(s)
		if s and s ~= "" and not seen[s] then seen[s] = true; table.insert(list, s) end
	end
	add((clean:gsub(" ", "_")))
	add(_llUrlEncode(clean))
	add((displayName:gsub(" ", "_")))
	add(_llUrlEncode(displayName))
	return list
end

local function _llFetchBody(requestFn, url)
	for attempt = 1, 3 do
		local ok, response = pcall(function()
			return requestFn({
				Url = url, Method = "GET",
				Headers = {
					["User-Agent"]      = "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36",
					["Accept"]          = "text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8",
					["Accept-Language"] = "en-US,en;q=0.5",
					["Cache-Control"]   = "no-cache",
				},
				Timeout = 10,
			})
		end)
		if ok and response and response.StatusCode == 200 and response.Body and response.Body ~= "" then
			return response.Body
		end
		if ok and response and response.StatusCode and response.StatusCode ~= 200 then
			return nil
		end
		if attempt < 3 then task.wait(0.5 * attempt) end
	end
	return nil
end

local function _llFetchFandomImage(displayName)
	if _LL_imgCache[displayName] ~= nil then
		return _LL_imgCache[displayName] or nil
	end
	local ok, body = pcall(function()
		local apiUrl = "https://stealabrainrot.fandom.com/api.php"
			.. "?action=query&prop=pageimages&format=json&pithumbsize=256&titles="
			.. (displayName:gsub(" ", "_"))
		local fn = _llRequestFn()
		if fn then
			local r = fn({ Url = apiUrl, Method = "GET",
				Headers = { ["User-Agent"] = "Mozilla/5.0" }, Timeout = 8 })
			return (r and r.StatusCode == 200) and r.Body or nil
		end
		return HttpService:GetAsync(apiUrl)
	end)
	if ok and body then
		local thumb = body:match('"source":"(https://[^"]+)"')
			or body:match('"url":"(https://static%.wikia%.nocookie%.net[^"]+)"')
		if thumb then
			thumb = thumb:gsub("\\/", "/"):gsub("&amp;", "&")
			_LL_imgCache[displayName] = thumb
			return thumb
		end
	end
	local requestFn = _llRequestFn()
	if requestFn then
		for _, slug in ipairs(_llWikiCandidates(displayName)) do
			local pageBody = _llFetchBody(requestFn, _LL_FANDOM_BASE .. slug)
			if pageBody then
				local og = pageBody:match('property="og:image"%s+content="([^"]+)"')
					or pageBody:match('content="([^"]+)"%s+property="og:image"')
					or pageBody:match('<meta%s+property="og:image"%s+content="([^"]+)"')
				if og and og ~= "" then
					og = og:gsub("&amp;", "&"):gsub("&quot;", '"')
					if og:find("^https?://") then
						_LL_imgCache[displayName] = og
						return og
					end
				end
				for _, attr in ipairs({ "data%-src", "src" }) do
					local last
					for _, ext in ipairs({ "png", "jpg", "jpeg", "webp" }) do
						local pat = attr .. '="(https://[^"]+%.' .. ext .. '[^"]*)"'
						for u in pageBody:gmatch(pat) do
							if u:find("static%.wikia%.nocookie%.net")
								and not u:find("/scale%-to%-width%-down/%d%d?$") then
								last = u
							end
						end
					end
					if last then
						last = last:gsub("/revision/latest", "")
						_LL_imgCache[displayName] = last
						return last
					end
				end
			end
		end
	end
	_LL_imgCache[displayName] = false
	return nil
end

end

local ws = nil
local function connectWebSocket()
	if ws then return end
	local success, sock = pcall(function()
		if WebSocket then return WebSocket.connect(WS_URL) end
		if syn and syn.websocket then return syn.websocket.connect(WS_URL) end
		if fluxus and fluxus.websocket then return fluxus.websocket.connect(WS_URL) end
		if websocket and websocket.connect then return websocket.connect(WS_URL) end
		return nil
	end)
	if not success or not sock then return end
	ws = sock
	ws.OnClose:Connect(function()
		ws = nil
		task.wait(1)
		connectWebSocket()
	end)
end

local function formatValue(num)
	if num >= 1e9 then
		return string.format("%.1fB", num / 1e9)
	elseif num >= 1e6 then
		return string.format("%.1fM", num / 1e6)
	elseif num >= 1e3 then
		return string.format("%.1fK", num / 1e3)
	else
		return tostring(num)
	end
end

local function getValue(text)
	if not text then return 0 end
	local num = tonumber(text:match("[%d%.]+")) or 0
	if text:find("[Mm]") then return num * 1000000
	elseif text:find("[Kk]") then return num * 1000
	elseif text:find("[Bb]") then return num * 1000000000
	else return num end
end

local isTeleporting = false
local function serverHop()
	if isTeleporting then
		print("[TricTrac Hop] Already teleporting — skipping duplicate call")
		return
	end
	if not hopEnabled then
		print("[TricTrac Hop] hopEnabled = false, skipping hop")
		return
	end
	isTeleporting = true
	print("\n[TricTrac Hop] === SERVER HOP ATTEMPT STARTED ===")
	print("[TricTrac Hop] Current JobId:", game.JobId, "| hopEnabled:", hopEnabled)
	pcall(function()
		if Player.Character and Player.Character:FindFirstChild("HumanoidRootPart") then
			Player.Character.HumanoidRootPart.Anchored = true
			print("[TricTrac Hop] Anchored HumanoidRootPart")
		end
	end)
	local placeId = game.PlaceId
	local currentJobId = game.JobId
	local success, rawData = pcall(function()
		return game:HttpGet("https://games.roblox.com/v1/games/" .. placeId .. "/servers/Public?sortOrder=Asc&limit=100")
	end)
	if not success then
		print("[TricTrac Hop] FAILED to HttpGet server list:", tostring(rawData))
		isTeleporting = false
		return
	end
	local servers
	success, servers = pcall(HttpService.JSONDecode, HttpService, rawData)
	if not success or not servers or not servers.data then
		print("[TricTrac Hop] FAILED to JSON decode server list")
		isTeleporting = false
		return
	end
	local validServers = {}
	for _, srv in ipairs(servers.data) do
		if srv.id ~= currentJobId and srv.playing < srv.maxPlayers then
			table.insert(validServers, srv)
		end
	end
	print("[TricTrac Hop] Found", #validServers, "valid servers out of", #servers.data, "total")
	if #validServers == 0 then
		print("[TricTrac Hop] No valid (different + non-full) servers found — will retry next cycle")
		isTeleporting = false
		return
	end
	local targetServer = validServers[math.random(1, #validServers)]
	print(string.format("[TricTrac Hop] Selected target → JobId: %s (%d/%d players)",
		targetServer.id, targetServer.playing, targetServer.maxPlayers))
	local tpSuccess, tpErr = pcall(function()
		print("[TricTrac Hop] Calling TeleportToPlaceInstance now...")
		TeleportService:TeleportToPlaceInstance(placeId, targetServer.id, Player)
	end)
	if tpSuccess then
		print("[TricTrac Hop] TeleportToPlaceInstance called SUCCESSFULLY (client-side)")
	else
		print("[TricTrac Hop] TeleportToPlaceInstance FAILED:", tostring(tpErr))
	end
	isTeleporting = false
	print("[TricTrac Hop] === HOP ATTEMPT FINISHED ===\n")
end

TeleportService.TeleportInitFailed:Connect(function(plr, teleportResult, errorMsg)
	if plr == Player then
		print("[TricTrac Hop] TELEPORT INIT FAILED →", teleportResult.Name, "| Message:", errorMsg)
		isTeleporting = false
		task.delay(3, function()
			if hopEnabled then
				print("[TricTrac Hop] Auto-retrying hop after TeleportInitFailed")
				serverHop()
			end
		end)
	end
end)

local lastAnimalData = {}

local function getAnimalHash(animalList)
	if not animalList then return "" end
	local hash = ""
	for slot, data in pairs(animalList) do
		if type(data) == "table" then
			hash = hash .. tostring(slot) .. tostring(data.Index) .. tostring(data.Mutation)
		end
	end
	return hash
end

local function scan()
	local items = {}
	local plots = Workspace:FindFirstChild("Plots")
	if not plots then return end

	for _, plot in ipairs(plots:GetChildren()) do
		pcall(function()
			local plotUID = plot.Name
			local channel = Synchronizer:Get(plotUID)
			if not channel then return end

			local animalList = channel:Get("AnimalList")
			local currentHash = getAnimalHash(animalList)
			if lastAnimalData[plotUID] == currentHash then return end
			lastAnimalData[plotUID] = currentHash

			local owner = channel:Get("Owner")
			if not owner or not Players:FindFirstChild(owner.Name) then return end

			if not animalList then return end

			for slot, animalData in pairs(animalList) do
				if type(animalData) == "table" then
					local animalName = animalData.Index
					local animalInfo = AnimalsData[animalName]
					if not animalInfo then continue end

					local mutation = animalData.Mutation or "None"
					local traits = (animalData.Traits and #animalData.Traits > 0) and table.concat(animalData.Traits, ", ") or "None"

					local genValue = AnimalsShared:GetGeneration(animalName, animalData.Mutation, animalData.Traits, nil)
					local genText = "$" .. NumberUtils:ToString(genValue) .. "/s"

					table.insert(items, {
						name = animalInfo.DisplayName or animalName,
						gen = genText,
						val = genValue,
						mutation = mutation,
						traits = traits,
					})
				end
			end
		end)
	end

	table.sort(items, function(a, b)
		return a.val > b.val
	end)

	if #items > 0 and items[1].val >= minValue then
		local bestItem = items[1]
		local logKey = bestItem.name .. "*" .. bestItem.gen .. "*" .. game.JobId
		if not loggedItems[logKey] then
			loggedItems[logKey] = true
			local extraBrainrots = {}
			local totalVal = bestItem.val
			for i = 2, math.min(#items, 8) do
				if items[i].val >= minValue then
					local extraLine = items[i].name .. " | " .. items[i].gen
					if items[i].mutation ~= "None" then
						extraLine = extraLine .. " | " .. items[i].mutation
					end
					table.insert(extraBrainrots, extraLine)
					totalVal = totalVal + items[i].val
				end
			end
			local players = #Players:GetPlayers() .. "/" .. Players.MaxPlayers
			sendWebhook(bestItem.name, bestItem.gen, bestItem.mutation, extraBrainrots, players, game.JobId, totalVal)
			sendWebSocket(bestItem.name, bestItem.gen, bestItem.mutation, players, game.JobId)
			print(string.format("[TricTrac] Found %s (%s) mutation=%s - value %s", bestItem.name, bestItem.gen, bestItem.mutation, bestItem.val))
		end
	end
end
